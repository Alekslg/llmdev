def inject_file_context(

    handler: Callable[[ModelRequest], ModelResponse]
) -> ModelResponse:
    """–í–Ω–µ–¥—Ä—è–µ—Ç –∫–æ–Ω—Ç–µ–∫—Å—Ç –æ —Ñ–∞–π–ª–∞—Ö, –∑–∞–≥—Ä—É–∂–µ–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–º –≤ —ç—Ç–æ–π —Å–µ—Å—Å–∏–∏."""

            {"role": "user", "content": file_context},
